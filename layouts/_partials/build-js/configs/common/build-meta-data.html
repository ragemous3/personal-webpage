{{/* purpose: Create a resource of variables to be fetched by runtime. Should ALWAYS be run last. inputs (.): - .pageParams (a .md files frontmatter params field) - .siteParams (values of config.tomls [params] or params.toml file) - worker_map (from Store) behavior: - Creates a file of config to be consumed by runtime. - Access store to fetch a map of fingerprinted workers output: file.json */}} {{- if not .pageParams -}} {{- errorf "Expected page params" -}} {{- end -}} {{- if not .siteParams -}} {{- errorf "Expected site params" -}} {{- end -}} {{- $payload := (dict "pageParams" .pageParams "siteParams" .siteParams "storeKey" "worker_map" ) }} {{- with (templates.Defer (dict "key" "runtime-config" "data" $payload )) }} {{- $workerMap := hugo.Store.Get "worker_map" | default (dict) }} {{- $json := (dict "workerMap" $workerMap ) | jsonify }} {{- $file := resources.FromString "data/runtime-config.json" $json }} {{- $file.Publish -}} {{- end }}
