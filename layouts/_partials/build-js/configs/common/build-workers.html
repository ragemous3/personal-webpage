{{/* partial: build-js/configs/common/build-workers purpose: Build a resource and then store its name under "workers" in hugo store output: void */}} {{- $workerKey := "workers" -}} {{- $workers := resources.Match "**/*.worker.ts" | default (slice) -}} {{- $sharedWorkers := resources.Match "**/*.shared-worker.ts" | default (slice) -}} {{- $allWorkers := slice | append $sharedWorkers | append $workers }} {{- range $res := $allWorkers -}} {{- $built := $res | js.Build (dict "format" "esm") -}} {{- $fileName := path.Base $res.Name -}} {{- if hugo.IsDevelopment -}} {{- hugo.Store.SetInMap $workerKey $fileName $built.RelPermalink -}} {{- else -}} {{- with ($built | fingerprint) -}} {{- hugo.Store.SetInMap $workerKey $fileName .RelPermalink -}} {{- end -}} {{- end -}} {{- end -}}
